SELECT *
FROM 
    (SELECT imdb_title_id AS title_id, title FROM movies) t1
 JOIN
    (SELECT imd_title_id AS title_id, imd_name_id AS name_id FROM title_principals) t2
ON (t1.title_id = t2.title_id);

FROM movies
LEFT JOIN title_principals;
